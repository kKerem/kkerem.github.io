Calling the photo uploaded specifically to the ID

function valuePhoto($photoID){
    $imageSrc = $photoID;
    if (is_numeric($photoID)) {
        $imageAttachment = wp_get_attachment_image_src($photoID,'full');
        $imageSrc = $imageAttachment[0];
        return $imageSrc;
    }
}
